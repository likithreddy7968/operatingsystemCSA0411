EXPERIMENT 12

#include <stdio.h>

int main() {
    int pages[50], frames[10];
    int n, f, i, j, page_faults = 0;
    int pointer = 0, flag;

    printf("Enter number of pages: ");
    scanf("%d", &n);

    printf("Enter the page reference string:\n");
    for (i = 0; i < n; i++)
        scanf("%d", &pages[i]);

    printf("Enter number of frames: ");
    scanf("%d", &f);

    // Initialize frames to -1 (empty)
    for (i = 0; i < f; i++)
        frames[i] = -1;

    printf("\nFIFO Page Replacement:\n");

    for (i = 0; i < n; i++) {
        flag = 0;

        // Check if page is already present
        for (j = 0; j < f; j++) {
            if (frames[j] == pages[i]) {
                flag = 1;    // Hit
                break;
            }
        }

        // If page not found â†’ Page Fault
        if (flag == 0) {
            frames[pointer] = pages[i];
            pointer = (pointer + 1) % f;
            page_faults++;
        }

        // Print current frame status
        printf("After inserting %d: ", pages[i]);
        for (j = 0; j < f; j++)
            printf("%d ", frames[j]);
        printf("\n");
    }

    printf("\nTotal Page Faults = %d\n", page_faults);

    return 0;
}

INPUT 

Enter number of pages: 7
Enter the page reference string:
1 3 0 3 5 6 3
Enter number of frames: 3

OUTPUT

FIFO Page Replacement:
After inserting 1: 1 -1 -1 
After inserting 3: 1 3 -1 
After inserting 0: 1 3 0 
After inserting 3: 1 3 0 
After inserting 5: 5 3 0 
After inserting 6: 5 6 0 
After inserting 3: 5 6 3 

Total Page Faults = 6
