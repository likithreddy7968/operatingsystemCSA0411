#include <stdio.h>
int main() {
    int n, i, j, bt[10], wt[10], tat[10], p[10], t;
    float avgWT=0, avgTAT=0;
    printf("Enter no. of processes: ");
    scanf("%d", &n);
    for(i=0;i<n;i++){ printf("BT of P%d: ",i+1); scanf("%d",&bt[i]); p[i]=i+1; }

    for(i=0;i<n-1;i++)
        for(j=i+1;j<n;j++)
            if(bt[i]>bt[j]){ t=bt[i];bt[i]=bt[j];bt[j]=t; t=p[i];p[i]=p[j];p[j]=t; }

    wt[0]=0;
    for(i=1;i<n;i++) wt[i]=wt[i-1]+bt[i-1];
    for(i=0;i<n;i++){ tat[i]=wt[i]+bt[i]; avgWT+=wt[i]; avgTAT+=tat[i]; }

    printf("\nP\tBT\tWT\tTAT\n");
    for(i=0;i<n;i++) printf("%d\t%d\t%d\t%d\n",p[i],bt[i],wt[i],tat[i]);
    printf("\nAvg WT=%.2f  Avg TAT=%.2f\n",avgWT/n,avgTAT/n);
}  

INPUT :

Enter no. of processes: 5
BT of P1: 2
BT of P2: 4
BT of P3: 46
BT of P4: 8
BT of P5: 9

OUTPUT :

P	BT	WT	TAT
1	2	0	2
2	4	2	6
4	8	6	14
5	9	14	23
3	46	23	69

Avg WT=9.00  Avg TAT=22.80
