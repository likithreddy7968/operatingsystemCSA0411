EXPERIMENT 6
ROUND ROBIN SCHEDULING 

#include <stdio.h>

int main() {
    int n, i, bt[10], rt[10], wt[10], tat[10], tq, time = 0, done;
    float avgWT = 0, avgTAT = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("Burst time of P%d: ", i + 1);
        scanf("%d", &bt[i]);
        rt[i] = bt[i];
    }

    printf("Enter time quantum: ");
    scanf("%d", &tq);

    do {
        done = 1;
        for(i = 0; i < n; i++) {
            if(rt[i] > 0) {
                done = 0;
                if(rt[i] > tq) {
                    time += tq;
                    rt[i] -= tq;
                } else {
                    time += rt[i];
                    wt[i] = time - bt[i];
                    rt[i] = 0;
                }
            }
        }
    } while(!done);

    for(i = 0; i < n; i++) {
        tat[i] = bt[i] + wt[i];
        avgWT += wt[i];
        avgTAT += tat[i];
    }

    printf("\nProcess\tBT\tWT\tTAT\n");
    for(i = 0; i < n; i++)
        printf("P%d\t%d\t%d\t%d\n", i + 1, bt[i], wt[i], tat[i]);

    printf("\nAverage Waiting Time = %.2f", avgWT / n);
    printf("\nAverage Turnaround Time = %.2f\n", avgTAT / n);

    return 0;
}

INPUT :

Enter number of processes: 4
Burst time of P1: 4
Burst time of P2: 5
Burst time of P3: 6
Burst time of P4: 7
Enter time quantum: 2

OUTPUT :

Process	BT	WT	TAT
P1	4	6	10
P2	5	12	17
P3	6	13	19
P4	7	15	22

Average Waiting Time = 11.50
Average Turnaround Time = 17.00
